{
	//[문제] 사용자의 원하는 수 만큼 int를 저장할 수 있는 메모리를 확보하라.
	//메모리의 내용을 1부터 시작하는 정수로 채워라.
	//합계를 계산해서 화면에 출력하라.

	size_t size;

	cin >> size;

	int* ptr;
	try {
		ptr = new int[size];
	}
	catch (exception& e)
	{
		cout << e.what() << endl;
		exit(404);
	}

	iota(ptr, ptr + size, 1);	// 1부터 시작하는 정수로 채우기

		sum += ptr[i];
